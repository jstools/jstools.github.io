jEngine: $template

Installation

npm install jstools-template --save

or

bower install jstools-template --save

Usage

var data = {
  foo: 'bar',
  crash: {
    test: 'dummy'
  },
  list: ['foo', 'bar', 'foobar'],
  map: {
    hi: 'all',
    bye: 'nobody'
  },
  template: 'sample',
  label: {
    cancel: 'cancel'
  }
};

$template('$if{ foo === "bar" }gogogo{:}whoops{/}')(data)
// returns 'gogogo'

$template('$each{ item, key in map }[${foo}:${key}:${item}]{/}')(data);
// returns '[bar:hi:all][bar:bye:nobody]'
