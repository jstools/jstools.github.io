jEngine: Events

Installation

npm install jengine-events --save

or

bower install jengine-events --save

Usage

var obj = new Events();

obj.on('foo', function () {
  flag = true;
});

obj.trigger('foo');

AngularJS Module

angular.module('myApp', ['jstools.events'])

  .controller('AppCtrl', ['$scope', 'Events', function ($scope, Events) {

    var obj = new Events();

    obj.$$on('foo', function () {
      flag = true;
    });

    obj.$$trigger('foo');

  }]);
